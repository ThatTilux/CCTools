189 // Generate timestamped log file name

192 std::stringstream ss;
193 ss << std::put_time(std::localtime(&now_time), "%Y-%m-%d_%H-%M-%S");
194 std::string timestamp = ss.str();
195 std::string log_filename = LOGS_DIR + "log_" + timestamp + ".txt";
196
197 // Create a multi-sink logger (console + file)
198 auto console_sink = std::make_shared<spdlog::sinks::stdout_color_sink_mt>();
199 auto file_sink = std::make_shared<spdlog::sinks::basic_file_sink_mt>(log_filename, true);
200
201 console_sink->set_pattern("[%Y-%m-%d %H:%M:%S] [%l] %v");
202 file_sink->set_pattern("[%Y-%m-%d %H:%M:%S] [%l] %v");
203
204 std::vector<spdlog::sink_ptr> sinks{console_sink, file_sink};
205
206 logger_ = std::make_shared<spdlog::logger>("multi_sink", begin(sinks), end(sinks));
207 spdlog::register_logger(logger_);
208
209 logger_->set_level(spdlog::level::debug); // Set to debug to log everything but trace
210 logger_->flush_on(spdlog::level::debug); // Flush on every log
